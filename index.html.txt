                    <input id="edit-isim" type="text" class="w-full bg-slate-50 border-2 border-slate-100 p-4 rounded-2xl focus:border-emerald-500 focus:outline-none font-bold text-slate-700 transition-all">

    <script>
        // Netlify üzerinden gelecek URL ve KEY
        // Kod içinde anahtar barýndýrmýyoruz, Netlify ayarlarýndan okuyacak
        const SB_URL = "https://mnagzuvwxvmzrlbzrfpp.supabase.co";
        const SB_KEY = "sb_publishable_Nzt_sPkSVU4pR9pc1hocmg_0Axrcv-W"; 
        const _supabase = supabase.createClient(SB_URL, SB_KEY);

        let aktifTab = 'gelir';
        let aktifCariId = null;

        window.onload = () => verileriGetir();

        function tabSec(tur) {
            aktifTab = tur;
            document.querySelectorAll('button[id^="tab-"]').forEach(btn => {
                btn.classList.replace('bg-emerald-600', 'bg-transparent');
                btn.classList.replace('text-white', 'text-slate-500');
            });
            const aktifBtn = document.getElementById(`tab-${tur}`);
            aktifBtn.classList.replace('bg-transparent', 'bg-emerald-600');
            aktifBtn.classList.replace('text-slate-500', 'text-white');
            verileriGetir();
        }

        async function verileriGetir() {
            const { data, error } = await _supabase.from('cariler').select('*').eq('tur', aktifTab).order('created_at', { ascending: false });
            const liste = document.getElementById('liste');
            
            if (data.length === 0) {
                liste.innerHTML = `<tr><td colspan="4" class="p-12 text-center text-slate-300 font-medium">Henüz kayýt bulunmuyor.</td></tr>`;
                return;
            }

            liste.innerHTML = data.map(item => `
                <tr class="group hover:bg-slate-50/50 transition-all">
                    <td class="p-6 font-bold text-slate-700">${item.isim}</td>
                    <td class="p-6 text-slate-400 text-sm font-medium">${item.kategori || 'Genel'}</td>
                    <td class="p-6 text-right">
                        <span class="bg-emerald-50 text-emerald-600 px-3 py-1 rounded-lg text-[10px] font-black uppercase tracking-tighter border border-emerald-100">Aktif</span>
                    </td>
                    <td class="p-6 text-center">
                        <button onclick="detayAc('${item.id}', '${item.isim}')" class="text-slate-300 group-hover:text-emerald-600 transition-colors">
                            <i class="fas fa-chevron-right"></i>
                        </button>
                    </td>
                </tr>
            `).join('');
        }

        function detayAc(id, isim) {
            aktifCariId = id;
            document.getElementById('edit-isim').value = isim;
            const m = document.getElementById('modal');
            m.classList.replace('hidden', 'flex');
        }

        function modalKapat() {
            const m = document.getElementById('modal');
            m.classList.replace('flex', 'hidden');
        }

        async function cariGuncelle() {
            const yeniIsim = document.getElementById('edit-isim').value;
            if(!yeniIsim) return;

            const { error } = await _supabase.from('cariler').update({ isim: yeniIsim }).eq('id', aktifCariId);
            if(!error) {
                modalKapat();
                verileriGetir();
            }
        }

        async function cariSil() {
            if(confirm('DÝKKAT: Bu kayýt ve bu kayda ait tüm geçmiţ silinecek. Emin misiniz?')) {
                const { error } = await _supabase.from('cariler').delete().eq('id', aktifCariId);
                if(!error) {
                    modalKapat();
                    verileriGetir();
                }
            }
        }

        async function yeniKayit() {
            const isim = prompt(`${aktifTab.toUpperCase()} adý giriniz:`);
            if (isim) {
                const { error } = await _supabase.from('cariler').insert([{ isim: isim, tur: aktifTab }]);
                if (error) alert("Hata: " + error.message);
                else verileriGetir();
            }
        }
    </script>
